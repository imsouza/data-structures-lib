774 Hufftree
775 createHufftree (char *string) {
776  Hufftree forest[FREQ];
777  int qty = 0;
778  int *leaf = frequency(string);
779 
780  for (int chr = 0; chr < FREQ; chr++) {
781  if (leaf[chr] > 0) {
782  insertTree(createHufftreeNode(NULL, chr, leaf[chr], NULL),
783  forest, &qty);
784  }
785  }
786 
787  while (qty > 1) {
788  Hufftree right = removeTree(forest, &qty);
789  Hufftree left = removeTree(forest, &qty);
790  insertTree(createHufftreeNode(left, '-', left->frq + right->frq, right),
791  forest, &qty);
792  }
793 
794  return forest[0];
795 }
